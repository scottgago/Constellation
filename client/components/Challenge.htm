function characterFrequency (string) {
  var index = 0
  var container = []
  var sortcontainer = {
    length: 0
  }
  var seen = {}
  for(var i = 0; i < string.length; i++){
  if(seen[string[i]] >= 0){
      container[seen[string[i]]][1]++
    } else {
      seen[string[i]] = index
      index++
      container.push([string[i], 1])
    }
  }
  for(var i = 0; i < container.length; i++){
    if(sortcontainer[container[i][1]]){
      sortcontainer[container[i][1]].push(container[i])
    } else {
      sortcontainer[container[i][1]] = [container[i]]
    }
  }
  
}
